def cifrar(texto):
    simbolos = "§×&0#12345678"
    cifrado = ""
    
    for caracter in texto:
        if caracter.isalpha():
            indice = ord(caracter) - ord('a')
            cifrado += simbolos[indice]
        elif caracter.isdigit():
            cifrado += simbolos[int(caracter) + 9]
        else:
            cifrado += caracter
            
    return cifrado

def descifrar(cifrado):
    simbolos = "§×&0#12345678"
    descifrado = ""
    
    for caracter in cifrado:
        if caracter in simbolos:
            indice = simbolos.index(caracter)
            if indice < 9:
                descifrado += chr(indice + ord('a'))
            else:
                descifrado += str(indice - 9)
        else:
            descifrado += caracter
            
    return descifrado

codigo_original = """

# Cifrar el código
codigo_cifrado = cifrar(codigo_original)
print(codigo_cifrado)

# Descifrar el código
codigo_descifrado = descifrar(codigo_cifrado)
print(codigo_descifrado)
